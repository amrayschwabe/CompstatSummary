datanew <-data[complete.cases(data),]

\begin{lstlisting}[language = R]
span <- c(0.1,0.2,0.3,0.45,0.7,1)
lo <- loess(y ~ x, span=span[i])}
predict(object=lo, x) # prediction
\end{lstlisting}
